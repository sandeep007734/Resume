\section{Current Research Projects}
\begin{itemize}

\item 
\normalsize{\textsc{SecureFS: A Secure File System for Intel SGX}} \\
\footnotesize{In this project, we are exploring an efficient and secure file system support for applications running in a secure enclave provided by Intel SGX, which does not support system calls, and consequently file system in its current form.}
\begin{flushright}
Advisor: \href{http://www.cse.iitd.ac.in/~srsarangi/}{Prof. Smruti R. \textsc{Sarangi}}
\end{flushright}

\item 
\normalsize{\textsc{Survey of Hardware assisted defense methods for CFI.}} \\
\footnotesize{We are working on a through analysis of hardware assisted defense methods to maintain control flow integrity of a binary's execution.}
\begin{flushright}
Advisor: \href{http://www.cse.iitd.ac.in/~srsarangi/}{Prof. Smruti R. \textsc{Sarangi}}
\end{flushright}

\item 
\normalsize{\textsc{Security aspects of a Remote Operations System or ROS}} \\
\footnotesize{ROS is widely used in autonomous vehicles. In this project we are exploring the current security challenges for an ROS based system and propose an efficient solution for the it.}
\begin{flushright}
Advisor: \href{http://www.cse.iitd.ac.in/~srsarangi/}{Prof. Smruti R. \textsc{Sarangi}}
\end{flushright}
